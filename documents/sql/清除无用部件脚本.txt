DELETE  from as_compo c where SUBSTR(C.COMPO_ID,0,2) IN(
SELECT DISTINCT TT FROM (
SELECT SUBSTR(C.COMPO_ID,0,2) TT FROM AS_COMPO C WHERE 
C.COMPO_ID NOT LIKE 'ZC%'
 AND C.COMPO_ID NOT LIKE 'GB%' 
AND C.COMPO_ID NOT LIKE 'AS%'
AND C.COMPO_ID NOT LIKE 'AP%'
AND C.COMPO_ID NOT LIKE 'EM%'
AND C.COMPO_ID NOT LIKE 'WF%'
  )
)
